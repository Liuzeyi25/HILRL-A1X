┌──────────────────────────────────────────────────────────┐
│  📋 工作流程                                              │
└──────────────────────────────────────────────────────────┘

  env.reset()
      │
      ├─ 🤖 机械臂 → 初始位置
      ├─ 🎮 Gello → 跟随机械臂 (2秒) ← 新功能！
      └─ ✅ 恢复遥控模式
      │
  env.step(action)
      │
      ├─ 👤 人类移动 Gello
      ├─ 🤖 机械臂跟随执行
      └─ 📊 记录演示数据

┌──────────────────────────────────────────────────────────┐
│  🚀 快速开始                                              │
└──────────────────────────────────────────────────────────┘

  # 录制演示（自动同步）
  python examples/record_demos_octo_manual.py \
    --exp_name a1x_pick_banana \
    --successes_needed 10 \
    --manual_success=True

┌──────────────────────────────────────────────────────────┐
│  ⚙️  配置参数                                              │
└──────────────────────────────────────────────────────────┘

  env = GelloIntervention(
      env,
      port="/dev/ttyUSB0",
      
      # 双向控制
      sync_on_reset=True,        # ← 启用Reset同步
      reset_follow_duration=2.0, # ← 同步时间
      
      # 遥控
      intervention_threshold=0.01
  )

┌──────────────────────────────────────────────────────────┐
│  🎯 核心功能                                              │
└──────────────────────────────────────────────────────────┘

  ✅ Reset同步    : Robot → Gello（自动）
  ✅ 遥控模式    : Gello → Robot（人工）
  ✅ 模式切换    : 全自动，无需操作
  ✅ 状态检测    : 智能提取关节位置

┌──────────────────────────────────────────────────────────┐
│  📝 使用示例                                              │
└──────────────────────────────────────────────────────────┘

  # 配置文件 (a1x_pick_banana/config.py)
  teleoperation_device = "gello"  # ← 已启用
  
  # 环境创建
  env = PickBananaEnv()
  env = GelloIntervention(env, port=gello_port)
  
  # 使用
  obs, _ = env.reset()  # ← Gello自动同步！
  obs, _, _, _, info = env.step(action)

  ├─ 检查: ls /dev/serial/by-id/
  ├─ 权限: sudo chmod 666 /dev/ttyUSB0
  └─ 测试: python examples/test_gello_bidirectional.py
  
  问题: 同步时间太短
  └─ 增加: reset_follow_duration=5.0
  
  问题: 想禁用同步
  └─ 设置: sync_on_reset=False
